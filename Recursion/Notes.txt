// This function demonstrates recursion by repeatedly calling itself
// with a smaller value until it reaches the base case (num === 0).
// Each call prints the current number, then decrements it by 1,
// creating a countdown from the initial value to 1.

        function fun(num) {
            if (num === 0) {
                return;
            }

            console.log(num);
            num -= 1;
            fun(num);
        }

        let a = 10;

        fun(a);

#1 Always try to make a recursion tree before making a recursion program.

#2 Another way of writing a recursion is, sum(n) = n + sum(n-1);

#3 Always make sure your recursive base case returns a valid value of the same type you're building (here, a number).
